94  private void GameControl_MouseDown(object sender, MouseButtonEventArgs e)

98 
99  int tpx = (int)tilePos.X;
100  int tpy = (int)tilePos.Y;
101 
102  if (tpx < this.model.Blocks.GetLength(0) &&
103  tpy < this.model.Blocks.GetLength(1) &&
104  this.model.Blocks[tpx, tpy] != null &&
105  this.model.Player != null &&
106  this.logic.IsNeighboring(tpx, tpy))
107  {
108  this.model.Blocks[tpx, tpy].DamageTaken(
109  this.model.Player.InflictDamage(),
110  this.model.Player);
111  if (this.model.Blocks[tpx, tpy].HealthPoints <= 0)
112  {
113  if (this.model.Blocks[tpx, tpy] as OneBlock != null)
114  {
115  this.logic.PlayerGainedMoney((this.model.Blocks[tpx, tpy] as OneBlock).Value);
116  }
117  else if (this.model.Blocks[tpx, tpy] as OneMonster != null)
118  {
119  this.logic.PlayerGainedMoney((this.model.Blocks[tpx, tpy] as OneMonster).Value);
120  }
121 
122  this.model.Blocks[tpx, tpy] = new OneBlock(tpx, tpy, BlockType.Air);
123  }
